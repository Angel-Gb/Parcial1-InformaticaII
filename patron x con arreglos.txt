#include <iostream>
using namespace std;

const int filas = 8; // Definición de la constante para el número de filas (8)
const int columnas = 8; // Definición de la constante para el número de columnas (8)

int main() {
    int patron1[filas][columnas]; // Declaración de una matriz llamada "patron1" de 8x8

    // Generar el patrón 1
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            // Esta condición verifica si (i, j) está en una de las dos diagonales de la matriz.
            // Si es así, establece el valor en (i, j) como 1, de lo contrario, lo establece como 0.
            if ((i == j) || (i + j == filas - 1)) {
                patron1[i][j] = 1;
            } else {
                patron1[i][j] = 0;
            }
        }
    }

    // Imprimir el patrón 1
    for (int i = 0; i < filas; i++) {
        for (int j = 0; j < columnas; j++) {
            cout << patron1[i][j]; // Imprime el valor en la posición (i, j) de "patron1"
            if (j < columnas - 1) {
                cout << ", "; // Agrega una coma y un espacio si no es la última columna.
            }
        }
        cout << endl; // Imprime un salto de línea después de imprimir una fila completa.
    }

    return 0;
}
