#include <iostream>
using namespace std;

int main() {
    int n = 8; // Número de columnas
    int m = 8; // Número de filas

    // Definición de dos secuencias de números (secuencia1 y secuencia2)
    int secuencia1[] = {1, 1, 0, 1, 1, 0, 1, 1};
    int secuencia2[] = {0, 1, 1, 0, 1, 1, 0, 1};

    // Iteración a través de las filas
    for (int i = 0; i < m; i++) {
        // Determina cuál secuencia se utilizará en esta fila basándose en el valor de i
        // Se alterna entre secuencia1 y secuencia2 en grupos de 4 filas.
        int* secuenciaActual = (i % 4 < 2) ? secuencia1 : secuencia2;

        // Iteración a través de las columnas
        for (int j = 0; j < n; j++) {
            cout << secuenciaActual[j] << " "; // Imprime el valor de la secuencia en la columna j
        }

        cout << endl; // Imprime un salto de línea al final de cada fila
    }

    return 0;
}
